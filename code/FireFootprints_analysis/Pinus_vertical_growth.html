normalize <- function(x) {
    return ((x - mean(x)) / sd(x))
  }

Re-create effects plot using predictorEffects() and ggplot

eff_shrubarea <- predictorEffect("shrubarea3_m", LM)
effects_df <- as.data.frame(eff_shrubarea)
ggplot(effects_df)+
  geom_line(aes(x = shrubarea3_m, y = fit))+
  geom_ribbon(aes(x = shrubarea3_m, ymin = lower, ymax = upper), alpha = .5, fill = "#fdbb84")+
  theme_bw()+
  xlab(bquote('Shrub cross-sectional area ' (m^2)))+
  ylab("Predicted relative growth (cm/cm)")+
  ggtitle("Pine growth in relation to shrub environment within 3 m")+
  geom_rug(data = df, aes(x = shrubarea3/10000, y = VertGrowth_Rel), alpha = .5, position = "jitter", sides = "b")+
